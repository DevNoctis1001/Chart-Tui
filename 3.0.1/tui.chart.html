    Example
    var chart = tui.chart; // or require('tui-chart');
var container = document.getElementById('container-id'),
    rawData = {
      categories: ['cate1', 'cate2', 'cate3'],
      series: [
        {
          name: 'Legend1',
          data: [20, 30, 50]
        },
        {
          name: 'Legend2',
          data: [40, 40, 60]
        },
        {
          name: 'Legend3',
          data: [60, 50, 10]
        },
        {
          name: 'Legend4',
          data: [80, 10, 70]
        }
      ]
    },
    options = {
      chart: {
        title: 'Area Chart'
      },
      yAxis: {
        title: 'Y Axis'
      },
      xAxis: {
        title: 'X Axis'
      }
    };
chart.areaChart(container, rawData, options);
